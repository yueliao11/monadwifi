**🔗 智能合约核心**
```solidity
function payAndReturn() external payable {
    uint256 returnAmount = (msg.value * 80) / 100;
    
    // 立即返还80%
    payable(msg.sender).transfer(returnAmount);
    
    // 记录支付历史
    emit PaymentMade(msg.sender, msg.value, returnAmount);
}
```
